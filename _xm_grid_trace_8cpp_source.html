304 void XmGridTraceImpl::TracePoint(const Pt3d& a_pt, const double& a_ptTime, VecPt3d& a_outTrace, VecDbl& a_outTimes)
305 {
306  m_exitMessage.clear();
307  double deltaT = 1.00;
308  double mag0 = 0, mag1 = 0;
309  Pt3d pt0 = a_pt, pt1;
310  double vx0 = 0, vx1 = 0, vy0 = 0, vy1 = 0, elapsedTime = 0;
311  bool bContinue = true;
312  Pt3d vtkVec; // Rename this variable
313  Pt3d vtkPt;
314  Pt3d vector;
315 
316  m_distTraveled = 0;
317  a_outTrace.clear();
318  a_outTimes.clear();
319  if (a_ptTime > m_time2 || //Test if the time specified is after the time range
320  !GetVectorAtLocationAndTime(a_pt,a_ptTime, vector)) // Ensure nothing fails during extraction
321  {
322  m_exitMessage = "Error occurred while extracting point0.";
323  return;
324  }
325  if (EQ_TOL(vector.x, XM_NODATA,1) || EQ_TOL(vector.y, XM_NODATA, 1))
326  {
327  m_exitMessage = "Point does not start inside an active cell.";
328  return;
329  }
330 
331  a_outTrace.push_back(a_pt);
332  a_outTimes.push_back(a_ptTime);
333 
334  vx0 = vector.x * m_vectorMultiplier;
335  vy0 = vector.y * m_vectorMultiplier;
336  mag0 = sqrt(vector.x * vector.x + vector.y * vector.y);
337 
338  double maxAngleChange = cos(m_maxChangeDirectionInRadians);
339 
340  while (bContinue)
341  {
342  if (m_maxChangeDistance > 0)
343  {
344  // make sure deltaT is small enough to not go past the max dist
345  double d2 = m_maxChangeDistance * m_maxChangeDistance;
346  double denom = (vx0 * vx0) + (vy0 * vy0) + (m_maxChangeDistance * XM_ZERO_TOL);
347  double dt = sqrt(d2 / denom);
348  if (deltaT > dt)
349  {
350  deltaT = dt;
351  m_exitMessage = "Change distance was greater than the max change distance.";
352  }
353  }
354  // If the change in DeltaT would push us beyond the time step, set it to hit the timestep
355  if (elapsedTime + deltaT + a_ptTime > m_time2)
356  {
357  deltaT = m_time2 - elapsedTime - a_ptTime;
358  bContinue = false; // This will be the last point traced
359  m_exitMessage = "The point has traveled beyond, or reached the second time step.";
360  }
361  // If the change in delta time would push beyond the max tracing time, set it to hit max tracing time
362  if (m_maxTracingTime > 0 && (elapsedTime + deltaT) > m_maxTracingTime)
363  {
364  deltaT = m_maxTracingTime - elapsedTime;
365  bContinue = false; // This will be the last point traced
366  m_exitMessage = "Exceeded or reached max tracing time.";
367  }
368 
369  // compute candidate point
370  pt1.x = pt0.x + deltaT * vx0;
371  pt1.y = pt0.y + deltaT * vy0;
372 
373  if (!GetVectorAtLocationAndTime(pt1, a_ptTime + elapsedTime + deltaT, vtkVec))
374  {
375  a_outTrace.clear();
376  a_outTimes.clear();
377  m_exitMessage = "Error occurred while extracting point1";
378  return;
379  }
380  // if pt1 outside of domain, compute new deltaT to get to boundary
381  if (EQ_TOL(vtkVec.x, XM_NODATA, 1) || EQ_TOL(vtkVec.y, XM_NODATA, 1))
382  {
383  m_exitMessage = "Point has traveled out of domain.";
384  VecPt3d points = { pt0,pt1 };
385  //DataLocationEnum is irrelevant here.
386  BSHP<XmUGrid2dPolylineDataExtractor> polylineExtractor = XmUGrid2dPolylineDataExtractor::New(m_ugrid, DataLocationEnum::LOC_POINTS);
387  polylineExtractor->SetPolyline(points);
388  points =polylineExtractor->GetExtractLocations();
389  if (points.size() < 3)
390  {
391  XM_LOG(xmlog::error, "Could not find intersection into no data area");
392  return;
393  }
394  double segDist = Mdist(pt0.x, pt0.y, pt1.x, pt1.y);
395  pt1 = points[points.size() - 2];
396  double newSegDist = Mdist(pt0.x, pt0.y, pt1.x, pt1.y);
397  deltaT *= (newSegDist / segDist);
398  bContinue = false;
399  if (!GetVectorAtLocationAndTime(pt1, a_ptTime + elapsedTime + deltaT, vtkVec)
400  || vtkVec.x == XM_NODATA || vtkVec.y == XM_NODATA)
401  {
402  m_exitMessage = "Error occurred while extracting point1";
403  return;
404  }
405  }
406  vx1 = vtkVec.x;
407  vy1 = vtkVec.y;
408  vx1 *= m_vectorMultiplier;
409  vy1 *= m_vectorMultiplier;
410 
411  if (EQ_TOL(vx1, 0.0, .0001) && EQ_TOL(vy1, 0.0, .0001)) //No velocity
412  {
413  a_outTrace.push_back(pt1);
414  a_outTimes.push_back(a_ptTime + elapsedTime + deltaT);
415  m_exitMessage = "Velocity has gone to zero.";
416  return;
417  }
418  bool bSplit = false;
419 
420  mag1 = sqrt(vx1 * vx1 + vy1 * vy1);
421 
422  // do we subdivide?
423 
424  if (!bSplit && m_maxChangeVelocity > 0)
425  {
426  double changeVel = fabs(mag1 - mag0);
427  if (changeVel > m_maxChangeVelocity)
428  {
429  bSplit = true;
430  m_exitMessage = "Point has exceeded max change velocity.";
431  }
432  }
433  if (!bSplit && m_maxChangeDirectionInRadians > 0)
434  {
435  double dir = getDirAsCosTheta(vx0, vy0, vx1, vy1);
436  if (dir < maxAngleChange)
437  {
438  bSplit = true;
439  m_exitMessage = "Point has exceeded max change direction.";
440  }
441  }
442  if (bSplit)
443  {
444  bContinue = true;
445  deltaT /= 2;
446  if (m_minDeltaTime>0 && deltaT < m_minDeltaTime) {
447  // done, exit
448  bContinue = false;
449  m_exitMessage += " Delta time was less than min delta time.";
450  }
451  }
452  else
453  {
454  double segDist = Mdist(pt0.x, pt0.y, pt1.x, pt1.y);
455  m_distTraveled += segDist;
456  if (m_maxTracingDistance != -1 && m_distTraveled > m_maxTracingDistance)
457  {
458  // because our last point exceeded the exitDistance
459  // find this point by linear calculations
460  double distancePast = m_distTraveled - m_maxTracingDistance;
461  double perc = distancePast / segDist;
462  Pt3d newPt;
463  newPt.x = (pt0.x * perc) + (pt1.x * (1 - perc));
464  newPt.y = (pt0.y * perc) + (pt1.y * (1 - perc));
465 
466  m_distTraveled = m_maxTracingDistance;
467  a_outTrace.push_back(newPt);
468  a_outTimes.push_back(a_ptTime + elapsedTime + deltaT*perc);
469  m_exitMessage = "Point has reached or exceeded the max tracing distance.";
470  return;
471  }
472 
473 
474  // add new pt if not identical to last
475  int size = (int)a_outTrace.size();
476  if (size > 0)
477  {
478  if (!EQ_TOL(pt1.x, a_outTrace.at(size - 1).x, XM_ZERO_TOL) ||
479  !EQ_TOL(pt1.y, a_outTrace.at(size - 1).y, XM_ZERO_TOL))
480  {
481  a_outTrace.push_back(pt1);
482  }
483  }
484  else
485  {
486  a_outTrace.push_back(pt1);
487  }
488  pt0 = pt1;
489  elapsedTime += deltaT;
490  vx0 = vx1;
491  vy0 = vy1;
492  deltaT *= 1.2;
493  mag0 = mag1;
494  a_outTimes.push_back(a_ptTime + elapsedTime);
495  }
496  } // while ()
